\newglossaryentry{rnaseq}{
  name=RNA-Seq,
  description={Transcriptome data from sequencing RNA}
}


\newglossaryentry{Sanger}{
  name=Sanger,
  description={A dideoxy chain termination method for DNA sequencing (named after Fred Sanger)}
}

\newglossaryentry{genomics}{
  name=genomics,
  description={An approach or technology designed to generate or use data from all genes in the genome},
  plural=genome %%switch
}

\newglossaryentry{omics}{
  name=omics,
  description={A rationale for applying molecular studies across all of the genes in the genome or biomolecules in the cell},
  plural=omic %%switch
}

\newglossaryentry{transcriptome}{
  name=transcriptome,
  description={An analysis of all the genes expressed in the genome},
  plural=transcriptomics
}

\newglossaryentry{metagenomics}{
  name=metagenomics,
  description={An analysis of all the genes and genomes in a community},
  plural=metagenome %%switch
}

\newglossaryentry{epigenome}{
  name=epigenome,
  description={An analysis of epigenetic modifications of all genes in the genome},
  plural=epigenomics
}

\newglossaryentry{proteome}{
  name=proteome,
  description={An analysis of all the proteins expressed from the genome},
  plural=proteomics
}

\newglossaryentry{metabolome}{
  name=metabolome,
  description={An analysis of all the metabolites and enzymes in the cell},
  plural=metabolomics
}

\newglossaryentry{methylome}{
  name=methylome,
  description={An analysis of DNA methylation throughout the genome},
  plural=methylomics
}

\newglossaryentry{exome}{
  name=exome,
  description={An approach or technology designed to generate or use data from all genes in the genome}
}

\newglossaryentry{bioinformatics}{
  name=bioinformatics,
  description={Statistical or computational approaches to biological data or research tools},
  plural=bioinformatic %%switch
}

\newglossaryentry{InDel}
{
  name=InDel,
  description={Changes to DNA sequence by insertion or deletion},
  plural=InDels
}

\newglossaryentry{microRNA}
{
  name=microRNA,
  description={Short RNA molecules generally regarded to regulate gene expression by binding to mRNA},
  plural=microRNAs
}

\newglossaryentry{synthetic lethal}
{
  name=synthetic lethality,
  description={Genetic interactions where inactivation of multiple genes is inviable (or deleterious) when they are viable if inactivated separately},
  plural=synthetic lethality
}
